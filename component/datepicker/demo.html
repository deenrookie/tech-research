参考文档

demo1

new datePicker(document.getElementById('datepicker1'));

demo2

自定义日期选择规则，本实例为自定义周末不能选择

new datePicker(document.getElementById('datepicker2'),{
	'validateMonth':function(y,m){
		if(m>10||m<6){
			return false;
		}
	},
	'customCell':function(date,elem){
		var day = date.getDay();
		if(day == 0||day == 6)
		this.disable(elem);
	}
});
