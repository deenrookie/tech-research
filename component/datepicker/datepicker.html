日期选择datapicker

引入js：http://qa.developer.163.com/component/datepicker/datepicker.js

默认css：http://qa.developer.163.com/component/datepicker/datepicker.css

javascript代码

var dp = new tinyDatepicker(elem,opt);
				
参数说明：
elem : 日期的输入域
opt : 配置项：
{
	temp : {
		row :'<tr>{0}</tr>', 			//行标签模版
		col :'<td>{0}</td>',				//列标签模版
		centent : ['<div><table>',
				'	<thead><tr>',
				'		<th>1</th><th>2</th><th>3</th>',
				'		<th>4</th><th>5</th><th>6</th><th>7</th>',
				'	</tr></thead>',
				'<tbody>{0}</tbody></table></div>'].join(""),	//日历主体容器模版
		title : ['<div class="datepicker_title">',
					'<a href="" class="next_btn"></a><a href="" class="prev_btn"></a>',
					'<span class="title_year">{year}</span>年',
					'<span class="title_month">{month}</span>月',
			 '</div>''].join("")			//日历顶部年和月显示
	},		
	format : 'yyyy/mm/dd'				//日期格式
	value : '2012/10/01'					//初始日期，格式要对于相应的日期格式，默认为当前日期
	custom : function(date,elem){			//自定义生成的表格元素节点
		//date 节点对于日期
		//elem 节点dom
	}
	cls : {							//自定义class名称
		'wrapper':'datepicker_wrapper',	//整体外部容器class
		'title':'datepicker_title', 			//顶部年月指示条
		'selector':'date_selector'			//主体的日期面板
	}
	validateMonth : function(year,month){	//控制月份是否可选
		if(month<=6||month>=10){
			return false;
		}
		return true;
	}
	customCell : function(date,elem){		//定制日期的dom节点
		if(date.getMonth()==11&&date.getDate()=='31')
		elem.innerHTML = '元旦';
	}
	showOtherMonth:true/false //前后补齐其他月份
}

dp = {
	formatDate:function(year,month,date){}//将给定的年,月,日格式化为指定的日期格式
	getOneMonth:function(year,month){}	//返回一个包含自定月份所有日期的数组
	prevMonth：						//显示上一个月
	nextMonth：						//显示下一个月
	show:							//显示日历组件
	hide:							//隐藏日历组件
	disable:function(elem){}				//将某一天的样式置为不可选
}
