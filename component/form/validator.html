验证说明

验证规则 验证项 代码参考
必填项 required _required = "true/false"
邮箱 email _type = "email"
网址 url _type = "url"
日期 date _type = "date"
ISO日期 dateISO _type = "dateISO"
数字 number _type = "number"
信用卡 creditcard _type = "creditcard"
两次输入验证 equalTo _type = "equalTo" _equalTo = "对比值的控件id"
文件类型限制 accept _type = "accept" _accept=".jpg,.png"
最大长度 maxlength _type = "maxlength" _maxlength="20"
最小长度 minlength _type = "minlength" _minlength="10"

最小值 mix _type = "mix" _mix="100"

var form = document.getElementById("form");	
var params = {
	autoCheck : true		//是否当输入项值发生改变时立即验证
	defaults : {
		errorClass: "st_form_warning",			//错误信息的样式名
		validClass: "st_form_valid",			//正确信息的样式名
		infoElement: "span"						//提示信息的标签
		}			
	messages : {
		required: "请输入必填项",
		email: "请输入正确的邮箱格式",
		url: "请输入正确的url",
		date: "请输入正确的日期",
		dateISO: "请输入一个合法的(ISO)时间",
		number: "请输入合法的数字",
		digits: "请输入整数",
		creditcard: "请输入合法的信用卡号",
		equalTo: "两次输入不一致，请重新输入",
		accept: "请输入正确后缀的值",
		maxlength: "请输入{1}个字符以下的内容",
		minlength: "请输入{0}个字符以上的内容",
		rangelength: "请输入长度在{0}和{1}之间的内容",
		range: "输入值必须在{0}与{1}之间",
		max: "输入值必须小于等于{1}",
		min: "输入值必须大于等于{0}",
		right : "信息输入正确"
		}
	methods : {

		}
};
