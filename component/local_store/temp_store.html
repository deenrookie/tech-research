调用初始化

			
NE._storage.init({
	talker:'swf_localstorage',	//as与js通信约定接口window['swf_localstorage']，可缺省
	targetId:'swf_node',	//必填  存放flash的容器id，切记不要添加影响flash运行的样式，如display:none;
	interval:3000,			//监听间隔（默认60s）
	needListen : true,		//是否需要监听，如调用addListener，必须设置true
	onready : function(){	//初始化加载完毕后，调用
	}				
});

接口调用说明

				
NE._storage.set(ns,key,value);					//存储，ns:命名空间
    
NE._storage.get(ns,key,function(params){			//取值，注：取值逻辑需要放到回调中（as异步时序）
	//params:(keyns, key, value)	
});	
	
NE._storage.remove(ns,key);                      			//清除，当key缺省时，清除ns下所有key
	
NE._storage.addListener(ns,key,function(params){	//添加监听
	//params:{keyns,key, newValue, oldValue}								
});
				
NE._storage.removeListener(ns,key);				//移除监听
