下拉框模拟selectmock

引入文件 http://qa.developer.163.com/component/select/select_mock.css

引入文件 http://qa.developer.163.com/component/select/select_mock.js

参数解释

node : 模拟的select对象
params : {
	widthDiff : 3, 		//宽度调整值，默认值为3
	maxHeight : 200,	//最大高度
	classes : {
		wrapperOpen : "c_select_mock_open",		//展开状态的样式
		wrapper : "c_select_mock"				//默认状态的样式
		selected : "selected"					//默认状态切换的样式
	},
	temp : {
		header : '<div class="c_select_arr">' + 
				'<input value="{text}" _value="{value}" readonly>' + 
				'<span></span></div><ul>',
		repeater : '<li lang="{value}" class="{selected}" _index="{index}">{text}</li>',
		footer : '</ul>'
	}
}

代码调用

var _select = new selectMock(node,params);

返回值

_select = {
	getNodes				//获取操作dom的方法，基于temp的改变需要做调整,
	getData				//数据定制
	createUI 				//生成UI的方法，返回值为html
	insertWrapper			//在select旁放入模拟控件
	attach				//（点击选择，滚动条自动定位，键盘支持）等交互事件处理
	toggle				//切换select的状态（打开或者关闭）
	setSelectData			//将用户选中的值进行存储操作
	getSelectedData		//获取用户选中的值并返回
	update				//更新下拉选项数据，可用于两级联动
	init : init(function(data){			//模拟执行方法,可传入回调，
		data = {
			value 		//select操作后的选中项值
			text			//select操作后的选中项文本
		}
	})		
}
