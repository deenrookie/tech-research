弹窗popOut

引入文件 http://qa.developer.163.com/component/pop/pop.css

引入文件 http://qa.developer.163.com/component/pop/pop.js

代码实例

var _pop = popOut(html,params);

参数解释：
	html : 弹窗内容
	
	params : {
		width : 500,			//弹窗宽度，默认500
		height : 'auto',			//弹窗高度，默认自适应
		left : 'auto',			//弹窗距离左侧的距离
		top : 'auto',			//弹窗距离顶部的距离
		mask : false,			//是否显示弹窗遮罩层，默认不显示
		zIndex : 300,			//弹窗的层级，默认300
		html : '',				//弹窗内容，默认为空
		temp : '<div class="common_popup common_popup_border">{html}</div>'
							//弹窗边框自定义，自定义{html}外面div，可自定义边框
	}
	
	_pop : {
		removeObj 			//删除子节点方法，传递需删除的节点
		popMask 				//生成遮罩层方法
		popClose 				//关闭弹窗方法
		nodes 				//访问弹窗Dom结构方法
		init 					//初始化弹窗
	}
