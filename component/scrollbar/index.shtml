js模拟滚动条

引入文件 http://qa.developer.163.com/component/scrollbar/scrollbar.css

引入文件 http://qa.developer.163.com/component/scrollbar/scrollbar.js

参数说明

 
//自定义参数
var opt = {
    "barId" : "textscrollbar", //滚动条的id名
    "barspacing" : 15, //滚动条跟内容之间的间距，默认为0
    "barAutoHeight" : true //滚动条是否随内容设置高，默认false固定高
};

代码调用

 
var bar = new ScrollBar();
bar.init(document.getElementById("main"));


var opt = {
    "barId" : "textscrollbar",
    "barspacing" : 15,
    "barAutoHeight" : true
};
var bar2 = new ScrollBar();
bar2.init(document.getElementById("main2"),opt);
